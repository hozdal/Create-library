## Install

```bash
npm install --save halil-btn-ui
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'halil-btn-ui'
import 'halil-btn-ui/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
